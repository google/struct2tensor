#### The BNF for a path is:


letter := [A-Za-z]
digit := [0-9]
<simple_step_char> := "_"|"-"| | letter | digit
<simple_step> := <simple_step_char>+
<extension> := "(" (<simple_step> ".")* <simple_step> ")"
<step> := <simple_step> | <extension>
<path> := ((<step> ".") * <step>)?
